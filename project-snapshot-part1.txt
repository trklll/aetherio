import { ReactNode, useRef, useEffect } from "react";
import { useLocation } from "react-router-dom";
import TopNav from "./TopNav";

const FULLBLEED_ROUTES = ["/home", "/detail", "/player", "/streams"];

export default function AppShell({ children }: { children: ReactNode }) {
  const loc       = useLocation();
  const hideNav   = loc.pathname === "/player" || loc.pathname === "/streams";
  const scrollRef = useRef<HTMLDivElement>(null);

  // Scroll to top al cambiar de ruta
  useEffect(() => {
    scrollRef.current?.scrollTo({ top: 0, behavior: "instant" as ScrollBehavior });
  }, [loc.pathname, loc.search]);

  const isFullbleed = FULLBLEED_ROUTES.some(r => loc.pathname.startsWith(r));

  return (
    <div style={{
      position: "relative",
      height: "100vh",
      width: "100vw",
      overflow: "hidden",
      background: "#1f1f1f",
      color: "#fff",
    }}>
      {!hideNav && <TopNav />}
      <div
        ref={scrollRef}
        style={{
          height: "100%",
          width: "100%",
          overflowY: "auto",
          overflowX: "hidden",
          paddingTop: hideNav || isFullbleed ? 0 : 72,
        }}
      >
        {children}
      </div>
    </div>
  );
}

import { useState, useRef, useEffect, useCallback } from "react";
import { useNavigate, useLocation } from "react-router-dom";
import { motion, AnimatePresence } from "framer-motion";
import { Search, X, Puzzle, Settings } from "lucide-react";
import clsx from "clsx";

const TMDB_KEY = "d8fbcec110971e64699f338caf0e7192";

const NAV_ITEMS = [
  { label: "Inicio",     to: "/home",             type: null },
  { label: "Películas",  to: "/home?type=movie",  type: "movie" },
  { label: "Series",     to: "/home?type=series", type: "series" },
  { label: "Anime",      to: "/home?type=anime",  type: "anime" },
  { label: "Biblioteca", to: "/library",           type: null },
];

interface Suggestion {
  id: number;
  title: string;
  media_type: "movie" | "tv";
  poster_path?: string;
  release_date?: string;
  first_air_date?: string;
}

export default function TopNav() {
  const navigate  = useNavigate();
  const location  = useLocation();
  const [searching, setSearching] = useState(false);
  const [query, setQuery]         = useState("");
  const [suggestions, setSuggestions] = useState<Suggestion[]>([]);
  const [showSugg, setShowSugg]   = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  // Active nav item detection
  const isActive = (to: string) => {
    const [path, qs] = to.split("?");
    const params = new URLSearchParams(qs ?? "");
    const type   = params.get("type");
    const curParams = new URLSearchParams(location.search);
    const curType = curParams.get("type");
    if (location.pathname !== path) return false;
    if (type) return curType === type;
    return !curType;
  };

  // Open search
  function openSearch() {
    setSearching(true);
    setTimeout(() => inputRef.current?.focus(), 300);
  }

  // Close search
  function closeSearch() {
    setSearching(false);
    setQuery("");
    setSuggestions([]);
    setShowSugg(false);
  }

  // Fetch suggestions (TMDB multi-search)
  const fetchSuggestions = useCallback(async (q: string) => {
    if (q.length < 2) { setSuggestions([]); setShowSugg(false); return; }
    try {
      const res  = await fetch(
        `https://api.themoviedb.org/3/search/multi?api_key=${TMDB_KEY}&query=${encodeURIComponent(q)}&language=es-ES&page=1`
      );
      const data = await res.json();
      const filtered = (data.results ?? [])
        .filter((r: any) => r.media_type === "movie" || r.media_type === "tv")
        .slice(0, 6);
      setSuggestions(filtered);
      setShowSugg(filtered.length > 0);
    } catch { setSuggestions([]); }
  }, []);

  function handleInput(val: string) {
    setQuery(val);
    if (debounceRef.current) clearTimeout(debounceRef.current);
    debounceRef.current = setTimeout(() => fetchSuggestions(val), 280);
  }

  function handleSearch(q = query) {
    if (!q.trim()) return;
    setShowSugg(false);
    navigate(`/search?q=${encodeURIComponent(q.trim())}`);
  }

  function handleSuggestion(s: Suggestion) {
    const type = s.media_type === "movie" ? "movie" : "series";
    const id   = `tmdb:${s.id}`;
    closeSearch();
    navigate(`/detail/${type}/${id}`);
  }

  // Escape to close
  useEffect(() => {
    const handler = (e: KeyboardEvent) => { if (e.key === "Escape") closeSearch(); };
    window.addEventListener("keydown", handler);
    return () => window.removeEventListener("keydown", handler);
  }, []);

  return (
    <div className="absolute top-0 left-0 right-0 z-50 flex justify-center pt-4" data-tauri-drag-region>
      <div className="relative liquid-glass-pill px-2 py-1.5 flex items-center min-w-0"
           style={{ boxShadow: "0 4px 24px rgba(0,0,0,0.5)" }}>

        {/* Logo */}
        <button
          onClick={() => { closeSearch(); navigate("/home"); }}
          className="w-8 h-8 rounded-full bg-white flex items-center justify-center shrink-0 mr-2 hover:scale-105 transition-transform"
        >
          <span className="text-black font-black text-sm">A</span>
        </button>

        {/* NAV ITEMS — ocultos durante búsqueda */}
        <AnimatePresence>
          {!searching && (
            <motion.div
              className="flex items-center gap-0.5"
              initial={{ opacity: 0, width: 0 }}
              animate={{ opacity: 1, width: "auto" }}
              exit={{ opacity: 0, width: 0 }}
              transition={{ duration: 0.25, ease: "easeInOut" }}
            >
              {NAV_ITEMS.map(({ label, to }) => (
                <button
                  key={to}
                  onClick={() => navigate(to)}
                  className={clsx(
                    "px-4 py-1.5 rounded-full text-sm font-medium transition-all duration-200 whitespace-nowrap",
                    isActive(to)
                      ? "bg-atv-selected text-white font-semibold"
                      : "text-atv-secondary hover:text-white hover:bg-atv-hover"
                  )}
                >
                  {label}
                </button>
              ))}
            </motion.div>
          )}
        </AnimatePresence>

        {/* SEARCH INPUT — aparece durante búsqueda */}
        <AnimatePresence>
          {searching && (
            <motion.div
              className="flex items-center"
              initial={{ opacity: 0, width: 0 }}
              animate={{ opacity: 1, width: 320 }}
              exit={{ opacity: 0, width: 0 }}
              transition={{ duration: 0.28, ease: [0.16, 1, 0.3, 1] }}
            >
              <Search size={15} className="text-atv-secondary shrink-0 ml-1 mr-2" />
              <input
                ref={inputRef}
                value={query}
                onChange={e => handleInput(e.target.value)}
                onKeyDown={e => e.key === "Enter" && handleSearch()}
                placeholder="Buscar películas, series, anime..."
                className="flex-1 bg-transparent text-sm text-white placeholder-atv-secondary focus:outline-none min-w-0"
              />
            </motion.div>
          )}
        </AnimatePresence>

        {/* Separador */}
        <div className="w-px h-4 bg-white/10 mx-2 shrink-0" />

        {/* SEARCH ICON / X */}
        <div className="flex items-center gap-1 shrink-0">
          {!searching ? (
            <button
              onClick={openSearch}
              className="p-2 rounded-full text-atv-secondary hover:text-white hover:bg-atv-hover transition-all duration-200"
            >
              <Search size={15} />
            </button>
          ) : (
            <button
              onClick={closeSearch}
              className="p-2 rounded-full text-atv-secondary hover:text-white hover:bg-atv-hover transition-all duration-200"
            >
              <X size={15} />
            </button>
          )}
          <button
            onClick={() => { closeSearch(); navigate("/addons"); }}
            className="p-2 rounded-full text-atv-secondary hover:text-white hover:bg-atv-hover transition-all duration-200"
            title="Add-ons"
          >
            <Puzzle size={15} />
          </button>
          <button
            onClick={() => { closeSearch(); navigate("/settings"); }}
            className="p-2 rounded-full text-atv-secondary hover:text-white hover:bg-atv-hover transition-all duration-200"
            title="Ajustes"
          >
            <Settings size={15} />
          </button>
        </div>
      </div>

      {/* SUGGESTIONS DROPDOWN */}
      <AnimatePresence>
        {searching && showSugg && suggestions.length > 0 && (
          <motion.div
            className="absolute top-[56px] liquid-glass-dark rounded-lg2 overflow-hidden min-w-[320px]"
            initial={{ opacity: 0, y: -8, scale: 0.97 }}
            animate={{ opacity: 1, y: 0, scale: 1 }}
            exit={{ opacity: 0, y: -8, scale: 0.97 }}
            transition={{ duration: 0.18, ease: "easeOut" }}
            style={{ boxShadow: "0 16px 48px rgba(0,0,0,0.7)" }}
          >
            {suggestions.map((s, i) => {
              const title = s.title ?? (s as any).name ?? "";
              const year  = (s.release_date ?? s.first_air_date ?? "").slice(0, 4);
              const typeLabel = s.media_type === "movie" ? "Película" : "Serie";
              return (
                <button
                  key={s.id}
                  onClick={() => handleSuggestion(s)}
                  className={clsx(
                    "w-full flex items-center gap-3 px-4 py-3 hover:bg-atv-hover transition-colors text-left",
                    i < suggestions.length - 1 && "border-b border-white/5"
                  )}
                >
                  {s.poster_path ? (
                    <img
                      src={`https://image.tmdb.org/t/p/w92${s.poster_path}`}
                      alt=""
                      className="w-8 h-12 object-cover rounded"
                    />
                  ) : (
                    <div className="w-8 h-12 bg-atv-surface rounded flex-shrink-0" />
                  )}
                  <div className="flex-1 min-w-0">
                    <p className="text-sm font-medium text-white truncate">{title}</p>
                    <p className="text-xs text-atv-secondary">{typeLabel}{year ? ` · ${year}` : ""}</p>
                  </div>
                  <Search size={13} className="text-atv-muted shrink-0" />
                </button>
              );
            })}
            <button
              onClick={() => handleSearch()}
              className="w-full px-4 py-3 text-sm text-atv-secondary hover:text-white hover:bg-atv-hover transition-colors text-left border-t border-white/5"
            >
              Ver todos los resultados de "<span className="text-white">{query}</span>"
            </button>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}

// Badges de formato estilo Apple TV+
// 4K · DV · HDR · Atmos · CC · SDH
// Se muestran siempre como indicadores de capacidad máxima de la plataforma

interface BadgeProps { label: string; style?: "box" | "text" }

function Badge({ label, style = "box" }: BadgeProps) {
  if (style === "text") {
    return (
      <span className="text-[10px] font-semibold text-atv-secondary tracking-wide">{label}</span>
    );
  }
  return (
    <span className="inline-flex items-center justify-center px-1 py-0.5 rounded text-[9px] font-bold tracking-wider border border-atv-secondary/50 text-atv-secondary leading-none">
      {label}
    </span>
  );
}

// Dolby Vision — styled badge
function DVBadge() {
  return (
    <span className="inline-flex items-center gap-0.5 px-1 py-0.5 rounded border border-atv-secondary/40 leading-none">
      <span className="text-[8px] font-black text-atv-secondary tracking-tighter">DOLBY</span>
      <span className="text-[8px] font-black text-atv-secondary tracking-tighter">VISION</span>
    </span>
  );
}

// Atmos
function AtmosBadge() {
  return (
    <span className="inline-flex items-center px-1 py-0.5 rounded border border-atv-secondary/40 leading-none">
      <span className="text-[8px] font-black text-atv-secondary tracking-tighter">ATMOS</span>
    </span>
  );
}

interface FormatBadgesProps {
  quality?: "4K" | "1080p" | "720p";
  showDV?: boolean;
  showAtmos?: boolean;
  showCC?: boolean;
  showSDH?: boolean;
}

export default function FormatBadges({
  quality = "4K",
  showDV = true,
  showAtmos = true,
  showCC = true,
  showSDH = true,
}: FormatBadgesProps) {
  return (
    <div className="flex items-center gap-1.5 flex-wrap">
      <Badge label={quality} />
      {showDV && <DVBadge />}
      {showAtmos && <AtmosBadge />}
      {showCC && <Badge label="CC" />}
      {showSDH && <Badge label="SDH" />}
    </div>
  );
}

function buildStreamId(query: StreamQuery) {
  if (query.type !== "movie" && query.season && query.episode) {
    return `${query.id}:${query.season}:${query.episode}`;
  }
  return query.id;
}

function normalizeStream(raw: any, addonId: string, addonName: string, index: number): MediaStream {
  const url = typeof raw.url === "string" ? raw.url : undefined;
  const externalUrl = typeof raw.externalUrl === "string" ? raw.externalUrl : undefined;
  const ytId = typeof raw.ytId === "string" ? raw.ytId : undefined;
  const infoHash = typeof raw.infoHash === "string" ? raw.infoHash : undefined;
  const title = typeof raw.title === "string" ? raw.title : undefined;
  const name = typeof raw.name === "string" ? raw.name : addonName;
  const fileIdx = typeof raw.fileIdx === "number" ? raw.fileIdx : undefined;
  const identity = [addonId, url, infoHash, title, fileIdx, index].filter(v => v !== undefined).join("|");

  return {
    id: identity,
    addonId,
    addonName,
    name,
    title,
    description: typeof raw.description === "string" ? raw.description : undefined,
    url,
    externalUrl,
    ytId,
    infoHash,
    fileIdx,
    behaviorHints: raw.behaviorHints,
    subtitles: Array.isArray(raw.subtitles) ? raw.subtitles : undefined,
  };
}

function dedupe(streams: MediaStream[]) {
  const seen = new Set<string>();
  return streams.filter(stream => {
    const key = stream.url ?? stream.externalUrl ?? stream.ytId ?? `${stream.infoHash ?? ""}:${stream.fileIdx ?? ""}:${stream.title ?? ""}`;
    if (!key || seen.has(key)) return false;
    seen.add(key);
    return true;
  });
}

export function useStreams(query: StreamQuery | null): UseStreamsResult {
  const getEnabledAddons = useAddonStore(s => s.getEnabledAddons);
  const [streams, setStreams] = useState<MediaStream[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [tick, setTick] = useState(0);

  const streamId = useMemo(() => query ? buildStreamId(query) : "", [query]);

  useEffect(() => {
    let cancelled = false;

    async function load() {
      if (!query || !streamId) return;
      setLoading(true);
      setError(null);

      const loaded: MediaStream[] = [];
      for (const addon of getEnabledAddons()) {
        const hasStreamResource = (addon.manifest?.resources ?? []).some((resource: any) => {
          if (typeof resource === "string") return resource === "stream";
          return resource?.name === "stream";
        });
        if (!hasStreamResource && addon.manifest?.resources?.length) continue;

        try {
          const base = addon.url.replace(/\/manifest\.json$/, "").replace(/\/$/, "");
          const response = await fetch(`${base}/stream/${query.type}/${encodeURIComponent(streamId)}.json`);
          if (!response.ok) continue;
          const json = await response.json();
          const addonStreams = (json.streams ?? []).map((stream: any, index: number) =>
            normalizeStream(stream, addon.id, addon.name, index)
          );
          loaded.push(...addonStreams);
        } catch {
          // One broken addon should not block the rest of the selector.
        }
      }

      if (!cancelled) {
        setStreams(dedupe(loaded));
        setLoading(false);
      }
    }

    load().catch(() => {
      if (!cancelled) {
        setError("No se pudieron cargar las fuentes.");
        setLoading(false);
      }
    });

    return () => { cancelled = true; };
  }, [getEnabledAddons, query, streamId, tick]);

  return {
    streams,
    loading,
    error,
    streamId,
    reload: () => setTick(value => value + 1),
  };
}

import { useAddonStore } from "../store/addonStore";

interface UseSubtitlesResult {

}

function buildVideoId(query: StreamQuery) {
  if (query.type !== "movie" && query.season && query.episode) {
    return `${query.id}:${query.season}:${query.episode}`;
  }
  return query.id;
}

function labelFor(raw: any) {
  return raw.label ?? raw.title ?? raw.lang ?? raw.language ?? "Subtítulos";
}

function normalizeSubtitle(raw: any, addonId: string, addonName: string, index: number): SubtitleSource | null {
  const url = typeof raw.url === "string" ? raw.url : undefined;
  if (!url) return null;
  const lang = raw.lang ?? raw.language ?? "und";
  return {
    id: [addonId, url, index].join("|"),
    addonId,
    addonName,
    url,
    lang,
    label: `${labelFor(raw)} · ${addonName}`,
    format: raw.format,
  };
}

function dedupe(items: SubtitleSource[]) {
  const seen = new Set<string>();
  return items.filter(item => {
    if (seen.has(item.url)) return false;
    seen.add(item.url);
    return true;
  });
}

export function useSubtitles(query: StreamQuery | null, stream?: MediaStream | null): UseSubtitlesResult {
  const getEnabledAddons = useAddonStore(s => s.getEnabledAddons);
  const [subtitles, setSubtitles] = useState<SubtitleSource[]>([]);
  const [loading, setLoading] = useState(false);
  const videoId = useMemo(() => query ? buildVideoId(query) : "", [query]);
  const subtitleLookup = useMemo(() => {
    if (!stream) return null;
    const behaviorHints = stream.behaviorHints ?? {};
    const videoHash = typeof behaviorHints.videoHash === "string" ? behaviorHints.videoHash : "";
    const videoSize = typeof behaviorHints.videoSize === "number" ? behaviorHints.videoSize : undefined;
    const filename = typeof behaviorHints.filename === "string" ? behaviorHints.filename : undefined;
    return { videoHash, videoSize, filename };
  }, [stream]);

  useEffect(() => {
    let cancelled = false;

    async function load() {
      if (!query || !videoId) {
        setSubtitles([]);
        setLoading(false);
        return;
      }
      setLoading(true);
      const loaded: SubtitleSource[] = [];

      for (const addon of getEnabledAddons()) {
        const hasSubtitles = (addon.manifest?.resources ?? []).some((resource: any) => {
          if (typeof resource === "string") return resource === "subtitles";
          return resource?.name === "subtitles";
        });
        if (!hasSubtitles) continue;

        try {
          const base = addon.url.replace(/\/manifest\.json$/, "").replace(/\/$/, "");
          const extraArgs = new URLSearchParams();
          extraArgs.set("videoID", videoId);
          if (subtitleLookup?.videoSize) extraArgs.set("videoSize", String(subtitleLookup.videoSize));
          if (subtitleLookup?.filename) extraArgs.set("filename", subtitleLookup.filename);
          if (subtitleLookup?.videoHash) extraArgs.set("videoHash", subtitleLookup.videoHash);
          const extraSegment = extraArgs.toString();
          const idsToTry = Array.from(new Set([
            subtitleLookup?.videoHash?.trim(),
            videoId,
          ].filter(Boolean) as string[]));
          const requestTargets = idsToTry.flatMap(id => {
            const encodedId = encodeURIComponent(id);
            return [
              extraSegment ? `${base}/subtitles/${query.type}/${encodedId}/${extraSegment}.json` : null,
              extraSegment ? `${base}/subtitles/${query.type}/${encodedId}.json?${extraSegment}` : null,
              `${base}/subtitles/${query.type}/${encodedId}.json`,
            ].filter(Boolean) as string[];
          });

          for (const target of requestTargets) {
            const response = await fetch(target);
            if (!response.ok) continue;
            const json = await response.json();
            const normalized = (json.subtitles ?? [])
              .map((subtitle: any, index: number) => normalizeSubtitle(subtitle, addon.id, addon.name, index))
              .filter(Boolean) as SubtitleSource[];
            loaded.push(...normalized);
            if (normalized.length > 0) break;
          }
        } catch {
          // Subtitle addons are optional; keep playback moving.
        }
      }

      if (!cancelled) {
        setSubtitles(dedupe(loaded));
        setLoading(false);
      }
    }

    load();
    return () => { cancelled = true; };
  }, [getEnabledAddons, query, subtitleLookup, videoId]);

  return { subtitles, loading };
}

function getStreamKind(stream: MediaStream): StreamKind {
  const url = stream.url?.toLowerCase() ?? "";
  if (url.startsWith("http://") || url.startsWith("https://")) return "https";
  if (stream.infoHash || url.startsWith("magnet:") || url.startsWith("stremio:")) return "p2p";
  if (url) return "external";
  return "unknown";
}

function looksWebPlayable(stream: MediaStream) {
  const target = (stream.url ?? stream.externalUrl ?? "").toLowerCase();
  const filename = String(stream.behaviorHints?.filename ?? "").toLowerCase();
  const title = `${stream.name ?? ""} ${stream.title ?? ""} ${stream.description ?? ""}`.toLowerCase();
  const source = `${target} ${filename} ${title}`;

  if (!target.startsWith("http://") && !target.startsWith("https://")) return false;
  if (/\.(m3u8|mpd|mkv|avi|mov|wmv|flv|ts|m2ts)(\?|$)/.test(target)) return false;
  if (/(x265|h265|hevc|10bit|remux|bluray)/.test(source)) return false;
  if (stream.behaviorHints?.notWebReady) return false;
  return /\.(mp4|m4v|webm|ogv|ogg)(\?|$)/.test(target) || /\.(mp4|m4v|webm|ogv|ogg)$/.test(filename);
}

function chooseEngine(stream: MediaStream, query: StreamQuery): PlaybackEngine {
  void query;
  if (looksWebPlayable(stream)) return "web";
  return "mpv";
}

function formatBytes(value?: number) {
  if (!value) return null;
  const gb = value / 1024 / 1024 / 1024;
  if (gb >= 1) return `${gb.toFixed(1)} GB`;
  return `${Math.round(value / 1024 / 1024)} MB`;
}

function buildPlayerSearch(params: URLSearchParams) {
  const next = new URLSearchParams();
  for (const key of ["type", "id", "season", "ep"]) {
    const value = params.get(key);
    if (value) next.set(key, value);
  }
  return next.toString();
}

function upgradeTmdbImage(url?: string) {
  if (!url) return url;
  return url.replace(/https:\/\/image\.tmdb\.org\/t\/p\/(?:w\d+|original)\//i, "https://image.tmdb.org/t/p/original/");
}

function normalizeMediaItem(item: MediaItem): MediaItem {
  return {
    ...item,
    poster: upgradeTmdbImage(item.poster),
    background: upgradeTmdbImage(item.background),
  };
}

export default function HomePage() {
  const [rows, setRows]           = useState<CatalogRowData[]>([]);
  const [loading, setLoading]     = useState(true);
  const [heroIndex, setHeroIndex] = useState(0);
  const location                  = useLocation();
  const getEnabled                = useAddonStore(s => s.getEnabledAddons);
  const typeFilter = new URLSearchParams(location.search).get("type");

  useEffect(() => { fetchAll(); }, []);

  const heroItems = rows.length > 0 ? rows[0].items.slice(0, 8) : [];
  useEffect(() => {
    if (heroItems.length < 2) return;
    const t = setInterval(() => setHeroIndex(i => (i + 1) % heroItems.length), 7000);
    return () => clearInterval(t);
  }, [heroItems.length]);

  async function fetchAll() {
    setLoading(true);
    const addons  = getEnabled();
    const allRows: CatalogRowData[] = [];
    for (const addon of addons) {
      for (const cat of addon.manifest?.catalogs ?? []) {
        try {
          const base = addon.url.replace(/\/manifest\.json$/, "").replace(/\/$/, "");
          const res  = await fetch(`${base}/catalog/${cat.type}/${cat.id}.json`);
          if (!res.ok) continue;
          const data  = await res.json();
          const seen  = new Set<string>();
          const items = (data.metas ?? [])
            .filter((m: any) => { if (seen.has(m.id)) return false; seen.add(m.id); return true; })
            .slice(0, 20)
            .map((item: MediaItem) => normalizeMediaItem(item));
          if (!items.length) continue;
          allRows.push({
            addonId: addon.id, addonName: addon.name,
            catalogId: cat.id, type: cat.type,
            name: cat.name ?? cat.id, items,
          });
        } catch { /* addon roto */ }
      }
    }
    setRows(allRows);
    setLoading(false);
  }

  const filteredRows = typeFilter ? rows.filter(r => r.type === typeFilter) : rows;

  if (loading) return <Skeleton />;
  if (!filteredRows.length) return <Empty typeFilter={typeFilter} />;

  const hero = heroItems[heroIndex];

  return (
    // fondo #1f1f1f — consistente con toda la app
    <div style={{ minHeight:"100vh", background:"#1f1f1f" }}>
      {hero && !typeFilter && (
        <HeroSection item={hero} items={heroItems} activeIndex={heroIndex} onSelect={setHeroIndex} />
      )}
      <div style={{ display:"flex", flexDirection:"column", gap:32, paddingBottom:48, background:"#1f1f1f" }}>
        {filteredRows.map((row, i) => (
          <CatalogRow key={`${row.addonId}-${row.catalogId}-${i}`} row={row} />
        ))}
      </div>
    </div>
  );
}

function getStreamKind(stream: MediaStream | null): StreamKind {
  const url = stream?.url?.toLowerCase() ?? "";
  if (url.startsWith("http://") || url.startsWith("https://")) return "https";
  if (stream?.infoHash || url.startsWith("magnet:") || url.startsWith("stremio:")) return "p2p";
  if (url) return "external";
  return "unknown";
}

function looksWebPlayable(stream: MediaStream | null) {
  if (!stream) return false;
  const target = (stream.url ?? stream.externalUrl ?? "").toLowerCase();
  const filename = String(stream.behaviorHints?.filename ?? "").toLowerCase();
  const title = `${stream.name ?? ""} ${stream.title ?? ""} ${stream.description ?? ""}`.toLowerCase();
  const source = `${target} ${filename} ${title}`;

  if (!target.startsWith("http://") && !target.startsWith("https://")) return false;
  if (/\.(m3u8|mpd|mkv|avi|mov|wmv|flv|ts|m2ts)(\?|$)/.test(target)) return false;
  if (/(x265|h265|hevc|10bit|remux|bluray)/.test(source)) return false;
  if (stream.behaviorHints?.notWebReady) return false;
  return /\.(mp4|m4v|webm|ogv|ogg)(\?|$)/.test(target) || /\.(mp4|m4v|webm|ogv|ogg)$/.test(filename);
}

function formatTime(value: number) {
  if (!Number.isFinite(value)) return "0:00";
  const total = Math.max(0, Math.floor(value));
  const hours = Math.floor(total / 3600);
  const minutes = Math.floor((total % 3600) / 60);
  const seconds = total % 60;
  if (hours) return `${hours}:${String(minutes).padStart(2, "0")}:${String(seconds).padStart(2, "0")}`;
  return `${minutes}:${String(seconds).padStart(2, "0")}`;
}

function getPlaybackTarget(stream: MediaStream) {
  return stream.url
    ?? stream.externalUrl
    ?? (stream.ytId ? `https://www.youtube.com/watch?v=${stream.ytId}` : undefined)
    ?? (stream.infoHash ? `magnet:?xt=urn:btih:${stream.infoHash}` : undefined)
    ?? "";
}

async function openExternal(stream: MediaStream, subtitle?: string) {
  const target = getPlaybackTarget(stream);
  if (!target) return { result: null, error: "La fuente no tiene URL reproducible." };
  if (target.toLowerCase().startsWith("magnet:")) {
    return {
      result: null,
      error: "Esta fuente es P2P/torrent y todavia necesita un motor de streaming antes de llegar a MPV.",
    };
  }
  try {
    const result = await invoke<MpvLaunchResult>("open_mpv", { target, subtitle });
    return { result, error: null };
  } catch (error) {
    return { result: null, error: String(error) };
  }
}

function buildQuery(params: URLSearchParams): StreamQuery | null {
  const type = params.get("type");
  const id = params.get("id");
  if (!type || !id) return null;
  const season = Number(params.get("season"));
  const episode = Number(params.get("ep"));
  return {
    type,
    id,
    season: Number.isFinite(season) && season > 0 ? season : undefined,
    episode: Number.isFinite(episode) && episode > 0 ? episode : undefined,
  };
}

async function resolveTmdbId(type: string, id: string) {
  if (id.startsWith("tmdb:")) return Number(id.slice(5));
  if (!id.startsWith("tt")) return null;
  const mediaType = type === "movie" ? "movie_results" : "tv_results";
  const response = await fetch(`${TMDB}/find/${id}?api_key=${TMDB_KEY}&external_source=imdb_id&language=es-ES`);
  if (!response.ok) return null;
  const json = await response.json();
  return json[mediaType]?.[0]?.id ?? null;
}
